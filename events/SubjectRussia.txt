		name = subject_rus.4.a

			limit = { check_variable = { SOV.subjectnevote < SOV.subjectvote } }
			set_country_flag = subject_aggresive_policy
			set_country_flag = SOV_campaign_war_net
			every_other_country = {
			limit = {
				has_autonomy_state = autonomy_republic_rf
				NOT = {has_idea = SUB_invervention_rus}
			}
			country_event = { id = subject_rus.2 days = 1 }
			}
		}
		if = {
			limit = { check_variable = { SOV.subjectnevote > SOV.subjectvote } }
			custom_effect_tooltip =  subject_nevotecampaign_tt
		}
		if = {
			limit = { check_variable = { SOV.subjectnevote = SOV.subjectvote } }
			custom_effect_tooltip =  subject_nevotecampaign_tt
		}
	}
