#European Belarus
country_event = {
	id = belarus_character.3
	title = belarus_character.3.t
	desc = belarus_character.3.d
	picture = GFX_era_sannikov

	is_triggered_only = yes

	option = {
		name = belarus_character.3.a
		log = "[GetDateText]: [This.GetName]: belarus_character.3.a executed"
		hidden_effect = {
			if = {
				limit = { has_idea = CSTO_member }
				leave_faction = yes
			}
			add_popularity = {
				ideology = communism
				popularity = -0.40
			}
			recalculate_party = yes
			if = { limit = { NOT = { is_in_array = { ruling_party = 2 } } }
				set_temp_variable = { rul_party_temp = 2 }
				change_ruling_party_effect = yes
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
				}
			}
			set_temp_variable = { party_index = 2 }
			set_temp_variable = { party_popularity_increase = 0.5 }
			set_temp_variable = { temp_outlook_increase = 0.07 }
			add_relative_party_popularity = yes
		}
		create_country_leader = {
			name = "Andrey Sannikov"
			picture = "Andrey_sannikov.dds"
			ideology = liberalism
			traits = {
				western_liberalism
			}
		}
	}
	option = {
		name = belarus_character.3.b
		log = "[GetDateText]: [This.GetName]: belarus_character.3.b executed"
		hidden_effect = {
			if = {
				limit = { has_idea = CSTO_member }
					leave_faction = yes
			}
			add_popularity = {
				ideology = communism
				popularity = -0.40
			}
			recalculate_party = yes
			if = { limit = { NOT = { is_in_array = { ruling_party = 1 } } }
				set_temp_variable = { rul_party_temp = 1 }
				change_ruling_party_effect = yes
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
				}
			}
				set_temp_variable = { party_index = 1 }
				set_temp_variable = { party_popularity_increase = 0.50 }
				set_temp_variable = { temp_outlook_increase = 0.07 }
				add_relative_party_popularity = yes
			}
	}
}
