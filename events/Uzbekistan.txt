add_namespace = UZB_karapalkstan
country_event = {
	id = UZB_karapalkstan.1 #Uzbeks and Karapalkans engage in conflict!
	title = UZB_karapalkstan.1.t
	desc = UZB_karapalkstan.1.d
	is_triggered_only = yes

	#Back the Karapalkans
	option = {
		name = UZB_karapalkstan.1.a
		log = "[GetDateText]: [This.GetName]: uzbekistan_event.1.a executed"
		custom_effect_tooltip = UZB_karapalkans_up_2
		add_to_variable = {
			var = UZB_karapalkans_opinion
			value = 2
		}
		custom_effect_tooltip = UZB_uzbeks_down_1
		add_to_variable = {
			var = UZB_uzbeks_opinion
			value = -1
		}
		ai_chance = {
			base = 20
		}
	}
	#Back the Uzbeks
	option = {
		name = UZB_karapalkstan.1.b
		log = "[GetDateText]: [This.GetName]: uzbekistan_event.1.a executed"
		custom_effect_tooltip = UZB_karapalkans_down_1
		add_to_variable = {
			var = UZB_karapalkans_opinion
			value = -1
		}
		custom_effect_tooltip = UZB_uzbeks_up_2
		add_to_variable = {
			var = UZB_uzbeks_opinion
			value = 2
		}
		ai_chance = {
			base = 10
		}
	}
	#Find a resolution for both sides
	option = {
		name = UZB_karapalkstan.1.c
		log = "[GetDateText]: [This.GetName]: uzbekistan_event.1.c executed"
		set_temp_variable = { treasury_change = -15 }
		modify_treasury_effect = yes
		add_political_power = -150
		custom_effect_tooltip = UZB_uzbeks_up_1
		add_to_variable = {
			var = UZB_uzbeks_opinion
			value = 1
		}
		custom_effect_tooltip = UZB_karapalkans_up_1
		add_to_variable = {
			var = UZB_karapalkans_opinion
			value = 1
		}
		ai_chance = {
			base = 10
		}
	}
}
