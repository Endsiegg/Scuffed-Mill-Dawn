country_event = {
	id = Turkey.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event Turkey.1" }

	title = Turkey.1.t
	desc = Turkey.1.desc
	picture = GFX_message_signing

	fire_only_once = yes

	is_triggered_only = yes

	option = {
		name = Turkey.1.a	#Let them go
		log = "[GetDateText]: [This.GetName]: Turkey.1.a executed"
		ai_chance = {
			factor = 80
		}
		set_temp_variable = { party_change = party_pop_array^3 }
		multiply_temp_variable = { party_change = 0.5 }
		set_temp_variable = { chp_boost = party_change }
		multiply_temp_variable = { chp_boost = 0.5 }
		set_temp_variable = { outlook_change = chp_boost }
		multiply_temp_variable = { outlook_change = -1 }
		#DSP loses half its MPs
		subtract_from_variable = { party_pop_array^3 = party_change }
		custom_effect_tooltip = TUR_DSP_LOSS_TT
		#Half of the lost MPs go to CHP
		add_to_variable = { party_pop_array^1 = chp_boost }
		custom_effect_tooltip = TUR_CHP_BOOST_TT
		#Adjust Outlook popularity
		add_popularity = { ideology = democratic popularity = outlook_change }
		recalculate_party = yes
	}
	option = {
		name = Turkey.1.b	#Backroom deals
		log = "[GetDateText]: [This.GetName]: Turkey.1.b executed"
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				is_historical_focus_on = yes
