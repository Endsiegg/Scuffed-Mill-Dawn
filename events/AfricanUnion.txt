country_event = {
	id = AfricanUnionRandomEvents.1
	title = AfricanUnionRandomEvents.1.t
	desc = AfricanUnionRandomEvents.1.d
	picture = GFX_african_union
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = African_Union_united }
		is_african_nation = yes
		has_idea = AU_member
		has_completed_focus = AFRICAN_UNION_shared_focus_create_investment_bank
		has_global_flag = African_Union_formed
		has_war = no
	}

	mean_time_to_happen = {
		#More likely to help large countries
		modifier = {
			factor = 1.25
			OR = {
				has_idea = regional_power
				has_idea = large_power
				has_idea = great_power
				has_idea = superpower
			}
		}
		modifier = {
			factor = 0.75
			has_idea = non_power
		}
		#Having good economy is plus
		modifier = {
			factor = 1.25
			OR = {
				has_idea = stable_growth
				has_idea = fast_growth
				has_idea = economic_boom
			}
		}
		modifier = {
			factor = 0.75
			OR = {
				has_idea = recession
				has_idea = depression
			}
		}
		#Low corruption good for investment
		modifier = {
			factor = 1.25
			OR = {
				has_idea = medium_corruption
				has_idea = modest_corruption
				has_idea = slight_corruption
				has_idea = negligible_corruption
			}
		}
		modifier = {
			factor = 0.75
			OR = {
				has_idea = unrestrained_corruption
				has_idea = rampant_corruption
				has_idea = crippling_corruption
				has_idea = paralyzing_corruption
			}
		}
		#Business oriented is good
		modifier = {
			factor = 1.25
			OR = {
				has_idea = small_medium_business_owners
				has_idea = industrial_conglomerates
				has_idea = maritime_industry
				has_idea = fossil_fuel_industry
			}
		}
		#Having banker connections is better
		modifier = {
			factor = 1.50
			has_idea = international_bankers
		}
		#Labour unions are bad for business
		modifier = {
			factor = 0.75
			has_idea = labour_unions
		}
		#Don't work with authoritarian countries
		modifier = {
			factor = 0.50
			OR = {
				has_idea = the_military
				has_idea = intelligence_community
				has_idea = foreign_jihadis
				has_elections = no
			}
		}
	}

	option = {
		name = AfricanUnionRandomEvents.1.a
		log = "[GetDateText]: [This.GetName]: AfricanUnionRandomEvents.1.a executed"
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				check_variable = { debt_ratio > 0.8 }
			}
		}
		set_temp_variable = { treasury_change = -3.75 }
		modify_treasury_effect = yes
		random_owned_state = {
			limit = {
				NOT = {
					has_state_category = state_inhospitable
					has_state_category = state_military_base
				}
			}
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			}
			add_extra_state_shared_building_slots = 1
		}
		random_other_country = {
			limit = {
				is_african_nation = yes
				has_idea = AU_member
			}
			set_temp_variable = { tag_index = THIS }
		}
		set_temp_variable = { influence_target = ROOT }
		set_temp_variable = { percent_change = 4 }
		change_influence_percentage = yes
	}
	option = {
		name = AfricanUnionRandomEvents.1.b
		log = "[GetDateText]: [This.GetName]: AfricanUnionRandomEvents.1.b executed"
		ai_chance = {
			factor = 20
		}
	}
