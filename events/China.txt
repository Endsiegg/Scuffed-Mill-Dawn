country_event = {
	id = china.9
	title = china.9.t
	desc = china.9.d
	picture = GFX_politics_salafism
	is_triggered_only = yes

	trigger = {
		original_tag = CHI
		CHI = {
			controls_state = 592
			check_variable = {
				var = uighur_opinion
				value = 100
				compare = less_than
			}
			check_variable = {
				var = uighur_threat
				value = 50
				compare = greater_than
			}
		}
	}


	option = {
		name = china.9.a
		log = "[GetDateText]: [This.GetName]: china.9.a executed"
		if = {
			limit = {
				original_tag = CHI
				check_variable = {
					var = uighur_threat
					value = 25
					compare = greater_than_or_equals
				}
				check_variable = {
					var = uighur_threat
					value = 50
					compare = less_than
				}
			}
			add_manpower = -25
			add_stability = -0.02
			set_temp_variable = { modify_uighur_threat = 1 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = {
				original_tag = CHI
				check_variable = {
					var = uighur_threat
					value = 50
					compare = greater_than_or_equals
				}
				check_variable = {
					var = uighur_threat
					value = 75
					compare = less_than
				}
			}
			add_manpower = -50
			add_stability = -0.03
			set_temp_variable = { modify_uighur_threat = 2 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = {
				original_tag = CHI
				check_variable = {
