			base = 1
			modifier = {
				factor = 1000
				has_global_flag = PRC_RED_DAWN_FOCUS_PATH
			}
		}
		log = "[GetDateText]: [This.GetName]: china.28.a executed"
	}

	option = {
		name = china.28.b #We can accept universal suffrage, as long as we vet the candidates first

			50 = {
				HKG = {
					swap_ideas = {

			}
			50 = {
				add_ideas = CHI_Hong_Kong_Protests_01

					set_country_flag = universal_suffrage_limited
					add_popularity = {
						ideology = nationalist
						popularity = 0.05

					add_ideas = Localist_Movement
					add_ideas = Hong_Kong_Protests_01

				news_event = hongkong_news.1

					clr_country_flag = universal_suffrage_limited
				}
			}
		}
		ai_chance = {
			base = 100
		}
		log = "[GetDateText]: [This.GetName]: china.28.b executed"
	}

	option = {
		name = china.28.c #The Hong Kong people should choose their leaders
		set_temp_variable = { temp_opinion = -5 }
		change_communist_cadres_opinion = yes
		add_popularity = {
			ideology = communism
			popularity = -0.05
		}
		recalculate_party = yes
		HKG = {
			if = {
				limit = { has_idea = Lacks_Universal_Suffrage }

					remove_idea = Lacks_Universal_Suffrage

				}
			}
			else_if = {
				limit = { has_idea = Limited_Universal_Suffrage }

					remove_idea = Limited_Universal_Suffrage

						check_variable = { influence_array^1 = CHI }
						check_variable = { influence_array_val^1 > 19.999 }

						check_variable = { influence_array^3 = CHI }
						check_variable = { influence_array_val^3 > 19.999 }
					}
				}
			}
			modifier = {
