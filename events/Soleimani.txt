country_event = {
	id = usa_soleimani.2
	title = usa_soleimani.2.t
	desc = usa_soleimani.2.d
	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				PER = {
					OR = {
						emerging_moderate_shiite_are_in_power = yes
						emerging_hardline_shiite_are_in_power = yes
					}
					PER_qasem_soleimani = { is_unit_leader = yes }
				}
				country_exists = USA
				USA = {
					NOT = {
						is_ally_with = PER
					}
				}
			}
			if = {
				limit = {
					country_exists = IRQ
				}
				country_event = usa_soleimani.1
			}
		}
	}
}

country_event = {
	id = usa_soleimani.1
	title = usa_soleimani.1.t
	desc = usa_soleimani.1.d

	picture = GFX_ghasem_soleimani_terror

	is_triggered_only = yes
	trigger = {
		OR = {
			emerging_hardline_shiite_are_in_power = yes
			emerging_moderate_shiite_are_in_power = yes
		}
	}
	option = { # Take a harsh revenge!
		name = usa_soleimani.1.a
		retire_character = PER_qasem_soleimani
		log = "[GetDateText]: [This.GetName]: usa_soleimani.1.a executed"
		unlock_decision_category_tooltip = PER_soleimani
		add_opinion_modifier = {
			target = USA
			modifier = assasinated_qasem_soleimani
		}
		add_named_threat = {
			threat = 5
			name = "Iran Attacks Ein Al-Asad Base"
		}

		hidden_effect = {
			news_event = {
				id = usa_soleimani_news.1
				days = 0
			}
			if = {
				limit = {
					557 = {
						is_fully_controlled_by = IRQ
					}
				}
				news_event = {
					id = usa_soleimani_news.2
					days = 1
				}
				country_event = {
					id = iranian_events.3
					days = 0
				}
			}
		}
		add_ideas = qasem_soleimani_spirit
		set_country_flag = harash_revenge
	}
	option = { #This means war
		name = usa_soleimani.1.b
		retire_character = PER_qasem_soleimani
		log = "[GetDateText]: [This.GetName]: usa_soleimani.1.b executed"
		add_opinion_modifier = {
			target = USA
			modifier = assasinated_qasem_soleimani
		}
		if = {
			limit = { country_exists = USA }
			declare_war_on = {
				target = USA
				type = puppet_wargoal_focus
			}
			add_named_threat = {
				threat = 10
				name = "Iran Declares War"
			}
		}
		hidden_effect = {
			news_event = {
				id = usa_soleimani_news.1
				days = 0
			}
			country_event = {
				id = iranian_events.3
				days = 0
			}
		}
		add_ideas = qasem_soleimani_spirit
	}
	option = { #Sad...
		name = usa_soleimani.1.c
		retire_character = PER_qasem_soleimani
		log = "[GetDateText]: [This.GetName]: usa_soleimani.1.c executed"
		add_opinion_modifier = {
			target = USA
			modifier = assasinated_qasem_soleimani
		}
		add_political_power = -150
		add_war_support = -0.05
		add_ideas = qasem_soleimani_spirit
		hidden_effect = {
			news_event = {
				id = usa_soleimani_news.1
				days = 0
			}
			country_event = {
				id = iranian_events.3
				days = 0
			}
		}
	}
}
