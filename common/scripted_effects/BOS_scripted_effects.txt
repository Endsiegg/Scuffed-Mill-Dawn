BOS_update_ethnic_tensions = {
	if = { limit = { check_variable = { bosniak_happiness > 3 } } set_variable = { var = croat_attitude value = 4 } }
	if = { limit = { check_variable = { bosniak_happiness > 3 } } set_variable = { var = srpska_attitude value = 4 } }
	if = { limit = { check_variable = { croat_happiness > 3 } } set_variable = { var = bosniak_attitude value = 4 } }
	if = { limit = { check_variable = { bosniak_happiness < 1 } } set_variable = { var = croat_attitude value = 1 } }
	if = { limit = { check_variable = { bosniak_happiness < 1 } } set_variable = { var = srpska_attitude value = 1 } }
	if = { limit = { check_variable = { croat_happiness < 1 } } set_variable = { var = bosniak_attitude value = 1 } }
}
