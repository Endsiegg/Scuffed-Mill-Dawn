modify_capitalisation_gazprom_support = {
	add_to_variable = { capitalisation_gazprom = modify_capitalisation_gazprom }
	custom_effect_tooltip = SOV_capitalisation_gazprom_tt
	clamp_variable = {
		var = capitalisation_gazprom
		min = 0.100
		max = 8000
	}
}
###########CAPITALISAION COMPANIES - GAZPROM CALCULATION EFFECT###########
capitalisation_gazprom_calculate = {
	if = {
		limit = {
			OR = {
				GER = {
					has_idea = SOV_nord_stream2_idea
					has_idea = SOV_nord_stream_idea
				}
				TUR = {
					has_idea = SOV_blue_stream_idea
					has_idea = SOV_turkey_stream_idea
				}
			}
		}
		SOV = {
			set_temp_variable = { modify_capitalisation_gazprom = 0.200 }
			modify_capitalisation_gazprom_support = yes
		}
	}
	if = {
		limit = {
			OR = {
				has_idea = depression
				has_idea = recession
				has_idea = stagnation
			}
		}
		SOV = {
			set_temp_variable = { modify_capitalisation_gazprom = -0.700 }
			modify_capitalisation_gazprom_support = yes
		}

				has_idea = fast_growth

			}
		}

		}

		}

###########CAPITALISAION COMPANIES - GAZPROM POSITIVE CHANGE EFFECT###########
capitalisation_gazprom_change_positive = {
	if = {
		limit = {
			has_idea = SOV_gazprom2_idea
			check_variable = { SOV.capitalisation_gazprom > 25 }
		}
		swap_ideas = {
			remove_idea = SOV_gazprom2_idea
			add_idea = SOV_gazprom3_idea
		}
	}
	else_if = {
		limit = {
			has_idea = SOV_gazprom1_idea
			check_variable = { SOV.capitalisation_gazprom > 12 }
		}
		swap_ideas = {
			remove_idea = SOV_gazprom1_idea
			add_idea = SOV_gazprom2_idea
		}
	}
	else_if = {
		limit = {
			has_idea = SOV_gazprom_idea
			check_variable = { SOV.capitalisation_gazprom > 3 }
		}
		swap_ideas = {
			remove_idea = SOV_gazprom_idea
			add_idea = SOV_gazprom1_idea
		}
	}
}

russian_tatarstan_annex = {
	if = {
		limit = {
			OR = {
				AND = {
					country_exists = TAT
					TAT = {
						is_subject_of = SOV
					}
				}
				SOV = { owns_state = 661 }
			}
			TAT = {
				NOT = { has_country_flag = SUB_subject_rebeliion_flag }
				NOT = { has_country_flag = SUB_subject_destroy_flag }
				check_variable = { TAT.subjectinped < 55 }
			}
		}
		add_political_power = -150
		add_stability = -0.05
		set_party_index_to_ruling_party = yes
		set_temp_variable = { party_popularity_increase = -0.08 }
		set_temp_variable = { temp_outlook_increase = -0.08 }
		add_relative_party_popularity = yes
		hidden_effect = {
			annex_country = { target = TAT transfer_troops = yes }
			add_to_variable = { treasury = TAT.treasury }
			add_to_variable = { debt = TAT.debt }
			add_to_variable = { int_investments = TAT.int_investments }
		}
		TAT = {
			every_core_state = { remove_core_of = TAT }
			set_country_flag = SUB_subject_rebeliion_flag
			set_country_flag = SUB_subject_destroy_flag
		}
	}
	else_if = {
		limit = {
			TAT = {
				NOT = { has_country_flag = SUB_subject_rebeliion_flag }
				NOT = { has_country_flag = SUB_subject_destroy_flag }
				check_variable = { TAT.subjectinped > 55 }
			}
			NOT = { country_exists = TAT }
		}
		if = {
			limit = {
				country_exists = TAT
				TAT = {
					is_subject_of = SOV
				}
			}
			country_event = { id = subject_rus.126 }
			SOV = {
				set_autonomy = {
					target = TAT
					autonomy_state = autonomy_free
				}
			}
		}
		else_if = {
			limit = {
				SOV = { owns_state = 661 }
			}
			release = TAT
			country_event = { id = subject_rus.126 }
		}
		TAT = {
			set_temp_variable = { modify_subjectinped = -25 }
			modify_subjectinped_support = yes
			set_country_flag = SUB_subject_rebeliion_flag
		}
	}
}
