		limit = {

		if = {
			limit = { check_variable = { UKR.easterns < UKR.westerns } }
			hidden_effect = {
				set_country_flag = ukr_revoult_bilo
				UKR = {
					country_event = { id = ukraine_maidan_west.1 days = 10 }
