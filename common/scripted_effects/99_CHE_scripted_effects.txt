CHE_chechen_start_create = {
  CHE = {	
			if = { limit = { NOT = { is_in_array = { ruling_party = 6 } } }
			set_temp_variable = { rul_party_temp = 6 }
			 change_ruling_party_effect = yes
			  set_politics = {
				 ruling_party = communism
				 elections_allowed = no
				 }
			 }
			 set_temp_variable = { party_index = 6 }
			 set_temp_variable = { party_popularity_increase = 0.7 }
			 set_temp_variable = { temp_outlook_increase = 0.6 }
			 add_relative_party_popularity = yes
			 swap_ideas = {
				remove_idea = defence_09
			 add_idea = defence_01
			}
			add_ideas = CHE_republic_rf
			add_ideas = CHE_kto
		}		
		CHE = { transfer_state = 672 }
}
