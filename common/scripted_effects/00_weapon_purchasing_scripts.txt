small_arms_USA = {
	set_temp_variable = { eq_amount = 240 }
	if = {
		limit = { USA = { has_tech = infantry_weapons7 } }
		set_temp_variable = { treasury_change = -15 }
		modify_treasury_effect = yes
		USA = {
			set_temp_variable = { treasury_change = 11.25 }
			modify_treasury_effect = yes
		}
		add_equipment_to_stockpile = {
			type = infantry_weapons7
			amount = eq_amount
			producer = USA
		}
