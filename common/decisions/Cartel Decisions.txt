cartels_decision_category = {
	narcos_crisis = { # Triggers a crisis when you reach level 10 corruption and you have drugs
		allowed = { always = no } #Disables Narco States
		icon = GFX_decision_cat_cartel

		visible = { #only visible to people with lv 10 corruption and if they have cartels already in the country
			OR = {
				has_idea = cartels_1
				has_idea = cartels_2
				has_idea = cartels_3
				has_idea = cartels_4
			}
			has_idea = paralyzing_corruption
			NOT = { TAG = MEX }
		}

		activation = {
			has_idea = paralyzing_corruption
			NOT = { TAG = MEX }
			OR = {
				has_idea = cartels_1
				has_idea = cartels_2
				has_idea = cartels_3
				has_idea = cartels_4
			}
		}

		selectable_mission = yes
		days_mission_timeout = 30

		timeout_effect = { #if they chose not to press the button the narcos rebel
			log = "[GetDateText]: [Root.GetName]: Decision timeout_effect narcos_crisis"
			start_civil_war = {
				ideology = neutrality
				size = 0.4
			}
		}

		complete_effect = { #if chosen the narcos take over
			log = "[GetDateText]: [Root.GetName]: Decision complete_effect narcos_crisis"
			swap_ideas = {
				remove_idea = paralyzing_corruption
				add_idea = negligible_corruption
			}
			every_neighbor_country = { #every neighbor's cartel level goes up by one
				if = {
					limit = { has_idea = cartels_3 }
					swap_ideas = {
						remove_idea = cartels_3
						add_idea = cartels_4
					}
				}
				else_if = {
					limit = { has_idea = cartels_2 }
					swap_ideas = {
						remove_idea = cartels_2
						add_idea = cartels_3
					}
				}
				else_if = {
					limit = { has_idea = cartels_1 }
					swap_ideas = {
						remove_idea = cartels_1
						add_idea = cartels_2
					}
				}
				else_if = {
					limit = {
						 NOT = { has_idea = narcos_state }
						 NOT = { has_idea = cartels_4 }
						 NOT = { has_idea = crime_syndicate_1 }
						 NOT = { has_idea = crime_syndicate_2 }
						 NOT = { has_idea = crime_syndicate_3 }
						 NOT = { has_idea = crime_syndicate_4 }
						 NOT = { has_idea = mafia_state }
						 NOT = { has_idea = ITA_mafia }
						}
					add_ideas = crime_syndicate_1
				}
			}
			start_civil_war = {
				ruling_party = neutrality
				ideology = ROOT
				size = 0.4
			}
			set_popularities = { #sets the Oligarchs in charge
                democratic = 0
                communism = 0
                fascism = 0.0
                neutrality = 100
                nationalist = 0
			}
			start_politics_input = yes
			set_variable = { party_pop_array^0 = 0 } #Western_Autocracy
            set_variable = { party_pop_array^1 = 0 } #conservatism
            set_variable = { party_pop_array^2 = 0 } #liberalism
            set_variable = { party_pop_array^3 = 0 } #socialism
            set_variable = { party_pop_array^4 = 0 } #Communist-State
            set_variable = { party_pop_array^5 = 0 } #anarchist_communism
            set_variable = { party_pop_array^6 = 0 } #Conservative
            set_variable = { party_pop_array^7 = 0 } #Autocracy
            set_variable = { party_pop_array^8 = 0 } #Mod_Vilayat_e_Faqih
            set_variable = { party_pop_array^9 = 0 } #Vilayat_e_Faqih
            set_variable = { party_pop_array^10 = 0 } #Kingdom
            set_variable = { party_pop_array^11 = 0 } #Caliphate
            set_variable = { party_pop_array^12 = 0 } #Neutral_Muslim_Brotherhood
            set_variable = { party_pop_array^13 = 0 } #Neutral_Autocracy
            set_variable = { party_pop_array^14 = 0 } #Neutral_conservatism
            set_variable = { party_pop_array^15 = 1 } #oligarchism
            set_variable = { party_pop_array^16 = 0 } #Neutral_Libertarian
            set_variable = { party_pop_array^17 = 0 } #Neutral_green
            set_variable = { party_pop_array^18 = 0 } #neutral_Social
            set_variable = { party_pop_array^19 = 0 } #Neutral_Communism
            set_variable = { party_pop_array^20 = 0 } #Nat_Populism
            set_variable = { party_pop_array^21 = 0 } #Nat_Fascism
            set_variable = { party_pop_array^22 = 0 } #Nat_Autocracy
            set_variable = { party_pop_array^23 = 0 } #Monarchist
            add_to_array = { ruling_party = 15 }
			startup_politics = yes
			set_country_leader_ideology = oligarchism

			remove_ideas = { #makes them a narco state
				cartels_1
				cartels_2
				cartels_3
				cartels_4
			}
			add_ideas = narcos_state

		}
