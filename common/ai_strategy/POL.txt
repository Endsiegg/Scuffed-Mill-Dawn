poland_hates_russia = {
	allowed = { original_tag = POL }
	enable = {
		country_exists = SOV
		# not emerging Poland
		NOT = { POL = { has_emerging_aligned_goverment = yes } }
		# do not hate neutral or pro-western Russia
		SOV = {
			OR = {
				has_emerging_aligned_goverment = yes
				has_nationalist_goverment = yes
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = contain id = "SOV" value = 25 }
	ai_strategy = { type = antagonize id = "SOV" value = 25 }
	ai_strategy = { type = befriend id = "SOV" value = -25 }
}
