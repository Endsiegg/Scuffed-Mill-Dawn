SUD_support_kablia = {
	allowed = {
		original_tag = SUD
		has_start_date < 2001.1.1
	}
	enable = {
		NOT = { has_war_with = DRC }
		DRC = { has_country_leader = { name = "Laurent-Dsir Kabila" } }
	}
	abort = {
		OR = {
			has_war_with = DRC
			DRC = { NOT = { has_country_leader = { name = "Laurent-Dsir Kabila" } } }
		}
	}

	ai_strategy = { type = befriend id = "DRC" value = 50 }
	ai_strategy = { type = support id = "DRC" value = 100 }
}
