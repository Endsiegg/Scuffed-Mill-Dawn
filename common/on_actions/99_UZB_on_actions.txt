on_actions = {
	on_monthly_UZB = {
		effect = {
			if = {
				limit = {
					country_exists = UZB
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
					}
				}
				uzbekistan_karakalpakstan_events = yes
			}
		}
	}
	on_weekly_UZB = {
		effect = {
			if = {
				limit = { check_variable = { UZB_karakalpakstans_opinion > 70 } }
				set_variable = { UZB_karakalpakstans_opinion = 70 }
			}
			if = {
				limit = {
					AND = {
						check_variable = {
							UZB_karakalpakstans_opinion > 50
							UZB_uzbeks_opinion > 50
					}
					UZB = { owns_state = 727 }
				}
				}
				UZB = { remove_ideas = UZB_regionalism_karalp }
			}
			#
			if = {
				limit = { check_variable = { UZB_karakalpakstans_opinion < 0 } }
				set_variable = { UZB_karakalpakstans_opinion = 0 }
			}
			#
			if = {
				limit = { check_variable = { UZB_uzbeks_opinion > 70 } }
				set_variable = { UZB_uzbeks_opinion = 70 }
			}
			#
			if = {
				limit = { check_variable = { UZB_uzbeks_opinion < 0 } }
				set_variable = { UZB_uzbeks_opinion = 0 }
			}
			# # # # # # # # # Will either demilitarize, or make a state independent if the opinion is too low # # # # # # # # #
			if = {
				limit = {
					check_variable = {
						UZB_karakalpakstans_opinion < 10
					}
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
						owns_state = 727
					}
				}
				727 = { set_demilitarized_zone = yes }
			}
			if = {
				limit = {
					check_variable = {
						UZB_karakalpakstans_opinion < 5
					}
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
						owns_state = 727
					}
				}
				727 = { set_demilitarized_zone = no }
				727 = { transfer_state_to = KRP }
				UZB = {
					set_country_flag = Karakalpaks_rebelled_yes
				}
				KRP = {
					declare_war_on = {
						target = UZB
						type = annex_everything
					}
					division_template = {
						name = "Karakalpak Militia"
						regiments = {
							Militia_Bat = { x = 0 y = 0 }
							Militia_Bat = { x = 0 y = 1 }
							Militia_Bat = { x = 1 y = 0 }
							Militia_Bat = { x = 1 y = 1 }
						}
						priority = 2
					}
					727 = {
						create_unit = {
							division = "name = \"Karakalpaks Militia\" division_template = \"Karakalpaks Militia\" start_experience_factor = 0.4"
							owner = KRP
						}
						create_unit = {
							division = "name = \"Karakalpaks Militia\" division_template = \"Karakalpaks Militia\" start_experience_factor = 0.4"
							owner = KRP
						}
						create_unit = {
							division = "name = \"Karakalpaks Militia\" division_template = \"Karakalpaks Militia\" start_experience_factor = 0.4"
							owner = KRP
						}
						create_unit = {
							division = "name = \"Karakalpaks Militia\" division_template = \"Karakalpaks Militia\" start_experience_factor = 0.4"
							owner = KRP
						}
					}
				}
			}
			if = {
				limit = {
					check_variable = {
						UZB_karakalpakstans_opinion > 10
					}
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
						owns_state = 727
					}
				}
				727 = { set_demilitarized_zone = no }
			}

			if = {
				limit = {
					check_variable = {
						UZB_uzbeks_opinion < 10
					}
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
						owns_state = 727
					}
				}
				727 = { set_demilitarized_zone = yes }
			}
			if = {
				limit = {
					check_variable = {
						UZB_uzbeks_opinion < 5
					}
					UZB = {
						NOT = {
							has_country_flag = Karakalpaks_rebelled_yes
						}
						owns_state = 727
					}
				}
				start_civil_war = {
					ideology = nationalist
					size = 0.1
				}
			}
		}
	}
}
